# @startpage / components

Unstyled React components for your startpage.

## Select

<SelectDemo />

### Props

| name        | type                       | description                                                                                           |
| ----------- | -------------------------- | ----------------------------------------------------------------------------------------------------- |
| className   | string \| undefined        | Pass your own css classes for unique styling.                                                         |
| disabled    | boolean \| undefined       | Disables the element.                                                                                 |
| options     | Option[]                   | Options of the select.                                                                                |
| placeholder | string \| undefined        | Placeholder if no option is selected.<br/> If a placeholder is passed, an empty option will be added. |
| value       | string \| undefined        | Current value of the element.                                                                         |
| leftExtra   | ReactFragment \| undefined | Component which is displayed on the left side.                                                        |
| rightExtra  | ReactFragment \| undefined | Component which is displayed on the right side.                                                       |
| onBlur      | () => void \| undefined    | Callback fired when the element loses focus.                                                          |
| onChange    | (value: string) => void    | Callback fired when the elements value changes.                                                       |
| onClose     | () => void \| undefined    | Callback fired when the content is collapsed.                                                         |
| onFocus     | () => void \| undefined    | Callback fired when the element gains focus.                                                          |
| onOpen      | () => void \| undefined    | Callback fired when the content is expanded.                                                          |

### Usage

```
import { useState } from "react"

import { faAngleDown } from "@fortawesome/free-solid-svg-icons"
import { FontAwesomeIcon } from "@fortawesome/react-fontawesome"
import { Option, Select } from "@startpage/components"

const options: Option[] = [
  {
    value: "test1",
    label: "Test 1",
  },
  {
    value: "test2",
    label: "Test 2",
  },
  {
    value: "test3",
    label: "Test 3",
  },
]

const Extra = <FontAwesomeIcon icon={faAngleDown} />

export const SelectDemo = () => {
  const [value, setValue] = useState<string>()

  const handleOpen = () => console.log("open!")
  const handleClose = () => console.log("close!")
  const handleChange = (value: string) => {
    setValue(value)
    console.log("value change!")
  }

  return (
    <Select
      value={value}
      options={options}
      rightExtra={Extra}
      onOpen={handleOpen}
      onClose={handleClose}
      onChange={handleChange}
      placeholder="Select a value"
    />
  )
}
```

## TabGroup & Tab

<TabDemo />

### Props

** TabGroup **

| name         | type                                 | description                                           |
| ------------ | ------------------------------------ | ----------------------------------------------------- |
| className    | string \| undefined                  | Pass your own css classes for unique styling.         |
| defaultIndex | number \| undefined                  | Tab which will be active on load.                     |
| vertical     | boolean \| undefined                 | Set this to true if you display your tabs vertically. |
| onChange     | (index: number) => void \| undefined | Callback which is fired when the active tab changes.  |

** Tab **

| name  | type   | description              |
| ----- | ------ | ------------------------ |
| title | string | Button title of the Tab. |

### SCSS template

```scss
div.stpg-tabgroup {
  > div.stpg-tabgroup-tablist {
    > button.stpg-tabgroup-tab {
      &[aria-selected="true"] {
      }
      :active {
      }
      :focus-visible {
      }
      :hover {
      }
    }
  }

  > div.stpg-tabgroup-panel {
  }
}
```

### Usage

```
import { Tab, TabGroup } from "@startpage/components"

export const TabDemo = () => {
  const handleChange = () => console.log("tab change!")

  return (
    <TabGroup
      defaultIndex={0}
      onChange={handleChange}
    >
      <Tab title="Tab 1">
        {content}
      </Tab>
      <Tab title="Tab 2">
        {content}
      </Tab>
      <Tab title="Tab 3">
        {content}
      </Tab>
    </TabGroup>
  )
}
```
