# @startpage/components

Unstyled React components for your startpage.

## Button

<ButtonDemo />

### Props

| name       | type                       | description                                     |
| ---------- | -------------------------- | ----------------------------------------------- |
| className  | string \| undefined        | Pass your own css classes for unique styling.   |
| disabled   | boolean \| undefined       | Disables the element.                           |
| leftExtra  | ReactFragment \| undefined | Component which is displayed on the left side.  |
| rightExtra | ReactFragment \| undefined | Component which is displayed on the right side. |
| onClick    | () => void \| undefined    | Callback fired when the element is clicked.     |

### SCSS template

```scss
button.stpg-button {
  > span.stpg-button-extra {
  }

  :active {
  }
  :focus-visible {
  }
  :hover {
  }
}
```

### Usage

```
import { faIcons } from "@fortawesome/free-solid-svg-icons"
import { FontAwesomeIcon } from "@fortawesome/react-fontawesome"
import { Button } from "@startpage/components"

const Extra = <FontAwesomeIcon icon={faIcons} />

export const ButtonDemo = () => {
  const handleClick = () => console.log("click!")

  return (
    <Button
      leftExtra={Extra}
      rightExtra={Extra}
      onClick={handleClick}
    >
      Click me!
    </Button>
  )
}
```

## Modal

<ModalDemo />

### Props

| name            | type                    | description                                       |
| --------------- | ----------------------- | ------------------------------------------------- |
| backdrop        | boolean \| undefined    | Adds an layer between modal and page content.     |
| className       | string \| undefined     | Pass your own css classes for unique styling.     |
| enterTransition | Transition \| undefined | Transitioning when modal enters the screen.       |
| leaveTransition | Transition \| undefined | Transitioning when modal leaves the scree.        |
| open            | boolean \| undefined    | Mounts / unmounts the modal.                      |
| onClose         | () => void              | Callback fired when a closing event is triggered. |

** Transition **

| name       | type   | description                                         |
| ---------- | ------ | --------------------------------------------------- |
| from       | string | CSS class applied before triggering the transition. |
| to         | string | CSS class applied to trigger the transition.        |
| transition | string | CSS class applied during the transition.            |

### SCSS template

```scss
div.stpg-modal {
  > div.stpg-modal-backdrop {
  }
  > div.stpg-modal-window {
    /* Pass those into the Transition object */
    &.hidden {
    }
    &.visible {
    }
    &.transition {
    }
  }
}
```

### Usage

```
import { useState } from "react"

import { Button, Modal } from "@startpage/components"

const enterTransition = {
  transition: "transition",
  from: "hidden",
  to: "visible",
}

const leaveTransition = {
  transition: "transition",
  from: "visible",
  to: "hidden",
}

export const ModalDemo = () => {
  const [open, setOpen] = useState(false)

  const handleOpen = () => {
    setOpen(true)
    console.log("open!")
  }
  const handleClose = () => {
    setOpen(false)
    console.log("close!")
  }

  return (
    <>
      <Button onClick={handleOpen}>Click me!</Button>
      <Modal
        backdrop
        enterTransition={enterTransition}
        leaveTransition={leaveTransition}
        open={open}
        onClose={handleClose}
      >
        {content}
      </Modal>
    </>
  )
}
```

## Select

<SelectDemo />

### Props

| name        | type                       | description                                                                                           |
| ----------- | -------------------------- | ----------------------------------------------------------------------------------------------------- |
| className   | string \| undefined        | Pass your own css classes for unique styling.                                                         |
| disabled    | boolean \| undefined       | Disables the element.                                                                                 |
| options     | Option[]                   | Options of the select.                                                                                |
| placeholder | string \| undefined        | Placeholder if no option is selected.<br/> If a placeholder is passed, an empty option will be added. |
| value       | string \| undefined        | Current value of the element.                                                                         |
| leftExtra   | ReactFragment \| undefined | Component which is displayed on the left side.                                                        |
| rightExtra  | ReactFragment \| undefined | Component which is displayed on the right side.                                                       |
| onBlur      | () => void \| undefined    | Callback fired when the element loses focus.                                                          |
| onChange    | (value: string) => void    | Callback fired when the elements value changes.                                                       |
| onClose     | () => void \| undefined    | Callback fired when the content is collapsed.                                                         |
| onFocus     | () => void \| undefined    | Callback fired when the element gains focus.                                                          |
| onOpen      | () => void \| undefined    | Callback fired when the content is expanded.                                                          |

### SCSS template

```scss
div.stpg-select {
  > button.stpg-select-header {
    > span.stpg-select-extra {
    }
    > span.stpg-select-text {
      &.placeholder {
      }
    }

    &[aria-expanded="true"] {
    }
    :focus-visible {
    }
    :hover {
    }
  }

  > ul.stpg-select-options {
    > li {
      &.active {
      }
      &.selected {
      }
    }
  }
}
```

### Usage

```
import { useState } from "react"

import { faAngleDown } from "@fortawesome/free-solid-svg-icons"
import { FontAwesomeIcon } from "@fortawesome/react-fontawesome"
import { Option, Select } from "@startpage/components"

const options: Option[] = [
  {
    value: "test1",
    label: "Test 1",
  },
  {
    value: "test2",
    label: "Test 2",
  },
  {
    value: "test3",
    label: "Test 3",
  },
]

const Extra = <FontAwesomeIcon icon={faAngleDown} />

export const SelectDemo = () => {
  const [value, setValue] = useState<string>()

  const handleOpen = () => console.log("open!")
  const handleClose = () => console.log("close!")
  const handleChange = (value: string) => {
    setValue(value)
    console.log("value change!")
  }

  return (
    <Select
      value={value}
      options={options}
      rightExtra={Extra}
      onOpen={handleOpen}
      onClose={handleClose}
      onChange={handleChange}
      placeholder="Select a value"
    />
  )
}
```

<SwitchDemo />

### Props

| name       | type                                  | description                                     |
| ---------- | ------------------------------------- | ----------------------------------------------- |
| checked    | boolean \| undefined                  | Checked state of the switch.                    |
| className  | string \| undefined                   | Pass your own css classes for unique styling.   |

| leftExtra  | ReactFragment \| undefined            | Component which is displayed on the left side.  |
| rightExtra | ReactFragment \| undefined            | Component which is displayed on the right side. |
| onBlur     | () => void \| undefined               | Callback fired when the element loses focus.    |
| onChange   | (value: boolean) => void \| undefined | Callback fired when the elements value changes. |
| onFocus    | () => void \| undefined               | Callback fired when the element gains focus.    |

### SCSS template

```scss
button.stpg-switch {
  > span.stpg-switch-extra {
  }
  > span.stpg-switch-knob {
  }
  > span.stpg-switch-track {
  }

  &[aria-checked="true"] {
  }
  :focus-visible > span.stpg-switch-track {
  }
}
```

      rightExtra={Extra}

## TabGroup & Tab

<TabDemo />

** TabGroup **

| name         | type                                 | description                                           |
| ------------ | ------------------------------------ | ----------------------------------------------------- |

      }
    }

  }

```

### Usage

```
import { Tab, TabGroup } from "@startpage/components"

export const TabDemo = () => {

## Text input

<TextInputDemo />

### Props

| name         | type                                                                                   | description                                                                                                        |
| ------------ | -------------------------------------------------------------------------------------- | ------------------------------------------------------------------------------------------------------------------ |
| autocomplete | string \| boolean \| undefined                                                         | Enables HTML basic autocomplete, pass a boolean for on / off or a string with the (HTML) type of the autocomplete. |
| autoFocus    | boolean \| undefined                                                                   | Input will be focused on load if set to true.                                                                      |
| className    | string \| undefined                                                                    | Pass your own css classes for unique styling.                                                                      |
| disabled     | boolean \| undefined                                                                   | Disables the element.                                                                                              |
| placeholder  | string \| undefined                                                                    | Placeholder which is shown if value is undefined.                                                                  |
| type         | "email" \| "number" \| "password" \| "search" \| "tel" \| "text" \| "url" \| undefined | Type of the input element, defaults to "text".                                                                     |
| value        | string \| undefined                                                                    | Current value of the element.                                                                                      |
| leftExtra    | ReactFragment \| undefined                                                             | Component which is displayed on the left side.                                                                     |
| rightExtra   | ReactFragment \| undefined                                                             | Component which is displayed on the right side.                                                                    |
| onBlur       | () => void \| undefined                                                                | Callback fired when the element loses focus.                                                                       |
| onChange     | (value: string) => void                                                                | Callback fired when the elements value changes.                                                                    |
| onFocus      | () => void \| undefined                                                                | Callback fired when the element gains focus.                                                                       |
| onKeyPress   | (key: string) => void \| undefined                                                     | Callback fired when a key is pressed while the input is focused.                                                   |

### SCSS template

```scss
div.stpg-text {
  > span.stpg-text-extra {
  }
  > input.stpg-text-input {
    ::placeholder {
    }

    :focus-visible {
    }
    :hover {
    }
  }
}
```

### Usage

```
import { useState } from "react"

import { faIcons } from "@fortawesome/free-solid-svg-icons"
import { FontAwesomeIcon } from "@fortawesome/react-fontawesome"
import { TextInput } from "@startpage/components"

const Extra = <FontAwesomeIcon icon={faIcons} />

export const TextInputDemo = () => {
  const [value, setValue] = useState("")

  const handleChange = (value: string) => {
    setValue(value)
    console.log("value change!")
  }
  const handleKeyPress = (key: string) => console.log(key)

  return (
    <TextInput
      placeholder="Placeholder"
      value={value}
      leftExtra={Extra}
      rightExtra={Extra}
      onChange={handleChange}
      onKeyPress={handleKeyPress}
    />
  )
}
```
